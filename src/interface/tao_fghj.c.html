179:    TaoComputeHessian() is typically used within minimization

189: @*/
190: int TaoComputeHessian(TAO_SOLVER solver,TaoVec *xx,TaoMat *HH)
191: {
192:   int    info;
193:   TaoTruth flg;

195:   TaoFunctionBegin;
196:   TaoValidHeaderSpecific(solver,TAO_COOKIE,1);
197:   if ( solver->taoappl){
198:     info = solver->taoappl->EvaluateHessian(xx,HH); CHKERRQ(info);
199:   }
200:   solver->nhesss++;

202:   info = HH->Compatible(xx,xx,&flg); CHKERRQ(info);
203:   if (flg==TAO_FALSE){

205:   }

207:   if (solver->viewhessian){
208:     info=HH->View();CHKERRQ(info);
209:   }

211:   TaoFunctionReturn(0);
212: }

586:          function and gradient evaluation routine (may be TAO_NULL)

588:    Calling sequence of func:
589: $     func (TAO_SOLVER solver,TaoVec *xx,double *f,void *ctx);

592: .  xx - variable vector
593: .  f - objective function value
594: -  ctx - [optional] user-defined function context

597: $     funcgrad (TAO_SOLVER solver,TaoVec *xx,double *f,TaoVec *gg, void *ctx);

603: -  ctx - [optional] user-defined function context

606: $     grad (TAO_SOLVER solver,TaoVec *xx,TaoVec *gg,void *ctx);

610: .  gg - gradient vector
611: -  ctx - [optional] user-defined function context 

613:    Calling sequence of destroy:
614: $     destroy (TAO_SOLVER solver, void *ctx);

813: /*@C
814:    TaoSetApplication - Sets the user defined context for 
815:    use by the optimization solver.  The application provides
816:    the solver with function and derivative information as
817:    well as data structures uses to store this information.

819:    Collective on TAO_SOLVER

821:    Input Parameters:
822: +  tao - the TAO_SOLVER solver context
823: -  taoapp - user application context

825:    Note:
826:    For PETSc users the TaoApplication* object is actually
827:    TAO_APPLICATION structure.

829:    
830:    Level: advanced

832: .keywords: application, context

834: @*/
835: int TaoSetApplication(TAO_SOLVER tao, TaoApplication *myapp){
836:   int info;
837:   TaoVec *xx,*RR;
838:   TaoMat *HH, *JJ;
839:   TaoVec *RXL, *RXU;
840:   TaoMat *CA;

842:   TaoFunctionBegin;
843:   tao->taoappl=myapp;
844:   info = myapp->GetVariableVector(&xx);CHKERRQ(info);
845:   info = myapp->GetHessianMatrix(&HH);CHKERRQ(info);
846:   info = myapp->GetJacobianMatrix(&JJ); CHKERRQ(info);
847:   info = myapp->GetConstraintVector(&RR);CHKERRQ(info);
848:   info = myapp->GetInequalityConstraints(&RXL,&CA,&RXU);

850:   tao->hessian=HH;
851:   tao->vec_sol=xx;
852:   tao->jacobian=JJ;
853:   tao->vfunc=RR;
854:   tao->RXL=RXL;
